Some cool random, helpful commands I can never remember since I don't use them that often.

# Hide or show hidden files / folders on a mac
defaults write com.apple.Finder AppleShowAllFiles YES
killall Finder

# Turn off shadows on screen captures 
defaults write com.apple.screencapture disable-shadow -bool false
killall SystemUIServer

# cURL a page repeatedly
for i in {1..1000}; do curl URL; done

# Show hidden files temporarily in Ubuntu 
CTRL + H

# Hide desktop icons in Ubuntu
gsettings set org.gnome.desktop.background show-desktop-icons false

# Apache error log location when attempting to fix .htacess problems 
# (updates as file changes occur)
tail -f /var/log/apache2/error.log

# Push to both the master branch and gh-pages branch on GitHub. Make similar changes in .git/config to the following lines.
[remote "origin"]
  url = git@github.com:capshaw/sudoku.git
	push = +refs/heads/master:refs/heads/gh-pages
	push = +refs/heads/master:refs/heads/master

# Simple static file server
python -m SimpleHTTPServer

# Fix multi-monitor behavior in Gnome
gsettings set org.gnome.shell.overrides workspaces-only-on-primary false
